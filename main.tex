% Labels
\newcommand{\highlight}[3]{%
    \tikz[baseline=(text.base)]{ \node[
        rectangle,
        rounded corners=14pt,
        fill=#1,
        text=#2,
        inner sep=10pt,
        outer sep=0pt,
        text width=, 
        align=left
        ] (text) {%
        \begin{varwidth}{\linewidth}
            \large \mybold{#3}
        \end{varwidth}
        };
    }%
}
